\section{Introduction}

\begin{frame}{Problem Statement}
\begin{itemize}
    \item Severe traffic congestion in Indian smart cities
    \item Existing traffic monitoring systems are:
    \begin{itemize}
        \item Expensive to deploy
        \item Limited in coverage
        \item Error-prone and manual
    \end{itemize}
    \item Need for a real-time, scalable, and cost-effective solution
\end{itemize}
\end{frame}

\begin{frame}{Proposed Solution}
AI-based intelligent traffic monitoring system using computer vision:
\begin{itemize}
    \item Real-time vehicle detection and counting
    \item Multi-object vehicle tracking
    \item Traffic density estimation
    \item Congestion detection and alerts
\end{itemize}
\bigskip
\textbf{Goal:} Assist smart city traffic management authorities
\end{frame}

\section{System Architecture}

\begin{frame}{System Overview}
\begin{itemize}
    \item \textbf{Input:} CCTV / IP camera video stream
    \item \textbf{Processing Pipeline:}
    \begin{enumerate}
        \item Vehicle Detection (YOLOv8)
        \item Multi-object Tracking (ByteTrack)
        \item Density and Congestion Analysis
        \item Visualization and Alerts
    \end{enumerate}
    \item \textbf{Output:} Vehicle counts, heatmaps, congestion alerts
\end{itemize}
\end{frame}

\begin{frame}{Key Technologies Used}
\begin{columns}[T]
    \column{0.5\textwidth}
    \begin{itemize}
        \item YOLOv8 (Nano model)
        \item OpenCV
        \item ByteTrack
        \item Python
    \end{itemize}
    \column{0.5\textwidth}
    \begin{itemize}
        \item NumPy, Pandas
        \item NVIDIA Jetson Nano / Laptop GPU
        \item 1080p CCTV Camera (30 FPS)
    \end{itemize}
\end{columns}
\end{frame}
